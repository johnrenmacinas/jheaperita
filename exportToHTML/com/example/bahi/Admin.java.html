package com.example.bahi;

import android.os.Bundle;

import androidx.annotation.NonNull;
import androidx.appcompat.app.AppCompatActivity;
import androidx.fragment.app.Fragment;
import androidx.fragment.app.FragmentManager;
import androidx.fragment.app.FragmentTransaction;

import com.example.bahi.databinding.ActivityAdminBinding;
import com.google.android.material.bottomnavigation.BottomNavigationView;
import com.google.firebase.auth.FirebaseAuth;

public class Admin extends AppCompatActivity {

    ActivityAdminBinding binding;
    private FirebaseAuth firebaseAuth;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        binding = ActivityAdminBinding.inflate(getLayoutInflater());
        setContentView(binding.getRoot());

        showHome_Admin();

        // Set up navigation item selection listener
        binding.bottomNavigationLandlord.setOnItemSelectedListener(item -> {
            // Get the ID of the menu item clicked
            int itemId = item.getItemId();
            if (itemId == R.id.bottom_home_landlord) {
                showHome_Admin();
                return true;
            } else if (itemId == R.id.bottom_listing_landlord) {
                showListing();
                return true;
            } else if (itemId == R.id.bottom_profile_landlord) {
                showProfile();
                return true;
            } else if (itemId == R.id.bottom_menu_landlord) {
                shoMenu();
                return true;
            } else {
                return false;
            }
        });

    }

    private void showHome_Admin() {

        Home_Admin fragment = new Home_Admin();
        FragmentTransaction fragmentTransaction = getSupportFragmentManager().beginTransaction();
        fragmentTransaction.replace(binding.fragmentContainerView2.getId(), fragment, "Home_Admin");
        fragmentTransaction.commit();
    }

    private void showListing() {

        Listing fragment = new Listing();
        FragmentTransaction fragmentTransaction = getSupportFragmentManager().beginTransaction();
        fragmentTransaction.replace(binding.fragmentContainerView2.getId(), fragment, "Listing");
        fragmentTransaction.commit();

    }

    private void showProfile() {

        Profile fragment = new Profile();
        FragmentTransaction fragmentTransaction = getSupportFragmentManager().beginTransaction();
        fragmentTransaction.replace(binding.fragmentContainerView2.getId(), fragment, "Profile");
        fragmentTransaction.commit();

    }

    private void shoMenu() {

        Menu_Landlord fragment = new Menu_Landlord();
        FragmentTransaction fragmentTransaction = getSupportFragmentManager().beginTransaction();
        fragmentTransaction.replace(binding.fragmentContainerView2.getId(), fragment, "Menu_Landlord");
        fragmentTransaction.commit();

    }

}
